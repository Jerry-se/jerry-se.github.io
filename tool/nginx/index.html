Nginx

首页: https://nginx.org 下载地址: https://nginx.org/en/download.html docs文档: http://nginx.org/en/docs/

有两种安装方式 1. 使用命令安装: http://nginx.org/en/linux_packages.html 2. 使用源码编译安装http://nginx.org/en/docs/configure.html 下载nginx pcre zlib wget https://nginx.org/download/nginx-1.20.1.tar.gz wget https://cfhcable.dl.sourceforge.net/project/pcre/pcre/8.43/pcre-8.43.tar.gz wget http://zlib.net/zlib-1.2.11.tar.gz 使用源码编译 ./configure --with-file-aio --with-http_ssl_module --with-http_stub_status_module --with-pcre=../pcre-8.43 --with-zlib=../zlib-1.2.11 编译参数意义 --with-http_stub_status_module: 主要用于查看Nginx的一些状态信息，参考文章 Nginx的源码编译安装与基础配置

默认的nginx.conf

user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log notice;
pid        /var/run/nginx.pid;


events {
    worker_connections  1024;
}


http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile        on;
    #tcp_nopush     on;

    keepalive_timeout  65;

    #gzip  on;

    include /etc/nginx/conf.d/*.conf;
}

[yl@localhost ~]$ vim /usr/lib/systemd/system/nginx.service

[Install]

配置 SSL 证书，配置文件 /etc/nginx/conf.d/deeplink.conf 如下:

server {
    listen 443 ssl;
    server_name www.deeplink.cloud deeplink.cloud;
    # ssl on;

    ssl_certificate   cert/deeplink.pem;
    ssl_certificate_key  cert/deeplink.key;
    ssl_session_timeout 5m;
    ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECDHE:ECDH:AES:HIGH:!NULL:!aNULL:!MD5:!ADH:!RC4;
    ssl_protocols TLSv1 TLSv1.1 TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;

    location / {
        if ($arg_invite ~* "^(\w+)$") {
            set $download_file "/DeeplinkSetup.exe";
            add_header Content-Disposition 'attachment; filename="DeeplinkSetup_$arg_invite.exe"';
        }
        root /data/dist;
        try_files $uri $uri/ /index.html;
        index index.html index.htm;
    }

    location = /50x.html {
        root html;
    }
    error_page 500 502 503 504  /50x.html;
}

server {
    listen 80;
    server_name deeplink.cloud www.deeplink.cloud;
    #return 301 http://$host$request_uri;
    #return 301 https://www.deeplink.cloud;
    return 301 https://${server_name}$request_uri;
}

配置 WebSocket 负载均衡如下:

http {

    upstream websocket_backend {
        ip_hash;
        server backend1.example.com:8080 weight=1 max_fails=2 fail_timeout=30s;
        server backend2.example.com:8080 weight=1 max_fails=2 fail_timeout=30s;
        # ... 可以添加更多的backend
    }

    server {
        listen 80;

        location /websocket {
            # proxy_pass http://127.0.0.1:19945;
            proxy_pass http://websocket_backend;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "Upgrade";
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        }
    }
}

其中： - ip_hash：确保基于客户端IP地址的请求始终被发送到相同的后端服务器，只要该服务器是可用的。 - max_fails 和 fail_timeout：定义一个后端服务器被认为是不可用之前可以有多少次失败，并且在被标记为失败后多久会重新检查。

fastcgi_index 语法：fastcgi_index file 默认值：none 使用字段：http, server, location 如果URI以斜线结尾，文件名将追加到URI后面，这个值将存储在变量$fastcgi_script_name中。例如： fastcgi_index index.php; fastcgi_param SCRIPT_FILENAME /home/www/scripts/php$fastcgi_script_name; 请求"/page.php"，参数SCRIPT_FILENAME将被设置为"/home/www/scripts/php/page.php"，但是请求"/"则为"/home/www/scripts/php/index.php"。
